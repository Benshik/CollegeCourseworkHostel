CREATE TABLE "Client" (
	"id"	INTEGER NOT NULL PRIMARY KEY AUTOINCREMENT,
	"FirstName"	TEXT NOT NULL,
	"SecondName"	TEXT NOT NULL,
	"MiddleName"	TEXT,
	"Birthday"	TEXT NOT NULL,
	"Gender"	TEXT NOT NULL
);



DROP VIEW VClientInformation;

CREATE VIEW VClientInformation As
SELECT 
 id,
 SecondName || " " || FirstName As FullName, 
 Birthday,
 Gender
FROM Client
ORDER by Birthday;

CREATE VIEW VReservesationInfo AS
SELECT 
Client.SecondName || " " || Client.FirstName As FullName,
Сouchette.Room,
Сouchette.idСouchette as numBed,
Reservation.Cost as costOrder
 FROM Reservation
 JOIN Client ON Client.id = Reservation.idClient
 JOIN Сouchette ON Сouchette.idСouchette = Reservation.idCouchete
